5 REM TASCHENRECHNER
10 INPUT "(P)LUS, (M)INUS, (G)ETEILT, ODER (MAL)";A$
20 INPUT "ERSTE ZAHL";B
30 INPUT "ZWEITE ZAHL";C
40 IF A$="P" THEN GOSUB 200
50 IF A$="M" THEN GOSUB 300
60 IF A$="G" THEN GOSUB 400
70 IF A$="MAL" THEN GOSUB 500
80 PRINT "DAS ERGEBNIS IST: ";D
90 INPUT "NOCH EINMAL (J/N)";E$
100 IF E$="J" THEN GOTO 10
110 IF E$="N" THEN STOP
200 REM PLUS
210 D=B+C
220 RETURN
300 REM MINUS
310 D=B-C
320 RETURN
400 REM GETEILT
410 D=B/C
420 RETURN
500 REM MAL
510 D=B*C
520 RETURN
